
cd ratcliff-group-resources/workflows/TADF_HTVS/FF_geopt_xyz_files_TXO-TPA

# create CSV header
echo "id,HOMO_eV,LUMO_eV" > txo-tpa_homo_lumo.csv

# extract values
for f in TXO-TPA_*_xtb.log; do
  id=${f#TXO-TPA_}
  id=${id%_xtb.log}
  awk -v id="$id" 'BEGIN{OFS=","}
    /\(HOMO\)/{h_eV=$(NF-1)}
    /\(LUMO\)/{l_eV=$(NF-1)}
    END{
      if(h_eV!="" && l_eV!=""){
        print id,h_eV,l_eV
      }
    }' "$f" >> txo-tpa_homo_lumo.csv
done

# quick preview
head txo-tpa_homo_lumo.csv
